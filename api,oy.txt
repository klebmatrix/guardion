def registrar_log(msg):
    agora = datetime.now().strftime("%H:%M:%S")
    logs.insert(0, f"[{agora}] {msg}")
    if len(logs) > 20: logs.pop()

async def sniper_loop():
    while True:
        if bot_config["status"] == "ON":
            try:
                async with httpx.AsyncClient(timeout=10.0) as client:
                    # Varre Polymarket e Esportes
                    await client.get("https://gamma-api.polymarket.com/events?active=true&limit=5")
                    registrar_log("Radar: Polymarket OK")
                    await client.get("https://api.dexsport.io/v1/events/active?chainId=137")
                    registrar_log("Radar: Esportes OK")
            except:
                registrar_log("Erro na varredura")
        await asyncio.sleep(15)

@app.on_event("startup")
async def startup():
    asyncio.create_task(sniper_loop())

@app.post("/entrar")
async def validar(pin: str = Form(...)):
    if pin == os.getenv("guardiao", "20262026"):
        return RedirectResponse(url="/dashboard", status_code=303)
    return "PIN INCORRETO"

@app.get("/dashboard", response_class=HTMLResponse)

@app.post("/toggle_bot")
async def toggle(status: str = Form(...)):
    bot_config["status"] = status
    registrar_log(f"Bot alterado para {status}")
    return RedirectResponse(url="/dashboard", status_code=303)
